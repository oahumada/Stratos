class SuccessionPlanningService
{
    /**
     * Calculate skill match score for person vs. target role
     * Returns: 0-100 (weighted score)
     */
    public function calculateSkillMatch(People $person, Roles $targetRole, Scenario $scenario): float
    {
        // 1. Get person's current skills + levels
        $personSkills = $person->skills()->pluck('pivot.proficiency_level', 'skill_id')->toArray();
        
        // 2. Get target role's required skills
        $requiredSkills = $targetRole->requiredSkills()->pluck('minimum_level', 'skill_id')->toArray();
        
        // 3. Calculate match: for each required skill, is person at level?
        $matchedCount = 0;
        $totalRequired = count($requiredSkills);
        foreach ($requiredSkills as $skillId => $minLevel) {
            if (isset($personSkills[$skillId)) && $personSkills[$skillId] >= $minLevel) {
                $matchedCount++;
            }
        }
        
        $baseScore = ($matchedCount / $totalRequired) * 100;
        
        // 4. Factor in: experience, training_in_flight, past_roles
        $experienceBonus = $this->calculateExperienceBonus($person, $targetRole);
        $trainingBonus = $this->calculateTrainingBonus($person, $scenario);
        
        return min(100, $baseScore + $experienceBonus + $trainingBonus);
    }

    /**
     * Assess readiness level based on skill match + experience + mentoring
     * Returns: "junior" | "intermediate" | "senior" | "expert"
     */
    public function assessReadiness(SuccessionCandidate $candidate): string
    {
        $score = $candidate->skill_match_score;
        $experience = $this->getRelevantExperienceYears($candidate->person, $candidate->targetRole);
        $mentorAvail = $this->getMentorAvailability($candidate); // 0-100

        $readinessScore = 
            ($score * 0.40) +           // 40% skill match
            (min(20, $experience * 5)) +  // 30% experience (capped at 20 points)
            ($mentorAvail * 0.20) +     // 20% mentor availability
            ($this->getMotivationSignal($candidate) * 0.10); // 10% motivation

        if ($readinessScore >= 85) return 'expert';
        if ($readinessScore >= 70) return 'senior';
        if ($readinessScore >= 50) return 'intermediate';
        return 'junior';
    }

    /**
     * Calculate succession coverage for all roles in scenario
     * Returns: [{role_id, role_name, covered_count, at_risk_count, critical_count, coverage_pct}]
     */
    public function calculateSuccessionCoverage(Scenario $scenario): array
    {
        $coverage = [];
        
        $roles = Roles::where('organization_id', $scenario->organization_id)
            ->where('is_critical', true) // Only critical roles
            ->get();

        foreach ($roles as $role) {
            $successors = SuccessionCandidate::where('scenario_id', $scenario->id)
                ->where('target_role_id', $role->id)
                ->get();

            $readyCount = $successors->where('readiness_level', 'expert')->count();
            $atRiskCount = $successors->whereIn('readiness_level', ['junior', 'intermediate'])->count();
            $criticalCount = $successors->filter(fn($s) => $s->skill_match_score < 50)->count();

            $coverage[] = [
                'role_id' => $role->id,
                'role_name' => $role->name,
                'covered_count' => $readyCount,
                'at_risk_count' => $atRiskCount,
                'critical_count' => $criticalCount,
                'coverage_pct' => $readyCount > 0 ? round(($readyCount / max(1, $successors->count())) * 100) : 0,
            ];
        }

        return $coverage;
    }

    // Helper methods (each ~50-100 LOC)
    private function calculateExperienceBonus(People $person, Roles $targetRole): float { ... }

    private function getMentorAvailability(SuccessionCandidate $candidate): float { ... }

}

class TalentRiskAnalyticsService
{
    /**
     * Analyze volatility risk for a person in organization context
     * Returns: 0-100 risk score
     */
    public function analyzeVolatilityRisk(People $person, Organization $organization): float
    {
        // Factors:
        $deptTurnoverRate = $this->getDepartmentTurnoverRate($person->department_id); // %
        $personTenure = $this->getPersonTenureYears($person); // years
        $roleCriticality = $person->currentRole->criticality ?? 5; // 1-10 scale
        $salaryVsMarket = $this->getSalaryCompetitiveness($person, $organization); // %

        $volatilityScore =
            ($deptTurnoverRate * 2) +           // Higher dept turnover = more risk (x2)
            (max(0, 10 - $personTenure) * 3) +  // Newer employees = more risk
            ((10 - $roleCriticality) * 5) +     // Less critical roles = higher flight risk
            (max(0, 100 - $salaryVsMarket) * 0.5); // Underpaid = risk

        return min(100, $volatilityScore);
    }

    /**
     * Predict retention probability for person in scenario context
     * Returns: 0-100 probability of staying
     */
    public function predictRetentionRisk(People $person, Scenario $scenario): float
    {
        $careerOpportunitiesInScenario = $this->asessCareerGrowth($person, $scenario); // 0-100
        $skillAlignmentWithFuture = $this->calculateFutureSkillFit($person, $scenario); // 0-100
        $teamDynamics = $this->assessTeamEngagement($person); // 0-100
        $compCompetitiveness = $this->getSalaryCompetitiveness($person); // 0-100

        $retentionScore =
            ($careerOpportunitiesInScenario * 0.30) +
            ($skillAlignmentWithFuture * 0.25) +
            ($teamDynamics * 0.20) +
            ($compCompetitiveness * 0.15) +
            (rand(0, 100) * 0.10); // Unknown factors buffer

        return min(100, $retentionScore);
    }

    /**
     * Identify skill gaps across workforce for scenario
     * Returns: [{skill_name, gap_count, priority, training_recommendations}]
     */
    public function calculateSkillGaps(Scenario $scenario): array
    {
        $gaps = [];
        
        // Get all skills needed in scenario future state
        $futureRoles = ScenarioRole::where('scenario_id', $scenario->id)->get();
        $requiredSkills = collect($futureRoles)
            ->flatMap(fn($sr) => $sr->role->requiredSkills)
            ->groupBy('skill_id')
            ->toArray();

        // Current workforce skills
        $currentSkills = People::where('organization_id', $scenario->organization_id)
            ->get()
            ->flatMap(fn($p) => $p->skills->pluck('pivot.proficiency_level', 'skill_id'))
            ->toArray();

        // Calculate gap
        foreach ($requiredSkills as $skillId => $requirements) {
            $currentLevel = $currentSkills[$skillId] ?? 0;
            $targetLevel = max($requirements->pluck('minimum_level')->toArray());
            
            if ($currentLevel < $targetLevel) {
                $gapSize = $targetLevel - $currentLevel;
                $peopleWithGap = People::where('organization_id', $scenario->organization_id)
                    ->whereHas('skills', fn($q) => $q->where('skill_id', $skillId)->where('proficiency_level', '<', $targetLevel))
                    ->count();

                $gaps[] = [
                    'skill_id' => $skillId,
                    'skill_name' => Skill::find($skillId)->name,
                    'gap_count' => $peopleWithGap,
                    'priority' => $gapSize >= 2 ? 'high' : 'medium',
                    'training_recommendations' => $this->recommendTrainingForSkill($skillId, $gapSize),
                ];
            }
        }

        return collect($gaps)->sortByDesc('gap_count')->values()->toArray();
    }

    // Helpers (~150 LOC)

    private function getPersonTenureYears(People $person): int { ... }

    private function asessCareerGrowth(People $person, Scenario $scenario): float { ... }
    private function calculateFutureSkillFit(People $person, Scenario $scenario): float { ... }

}

class TransformationRoadmapService
{
    /**
     * Generate phased transformation roadmap from current to future state
     */
    public function generateTransformationPlan(Scenario $scenario, array $options = []): void
    {
        // Delete existing phases for scenario (fresh generation)
        TransformationPhase::where('scenario_id', $scenario->id)->delete();

        $totalTimelineMonths = $scenario->timeline_weeks / 4;

        // Phase 1 (0-6 months): Foundation
        $this->createPhase($scenario, 1, 'Foundation', 0, 6, [
            'objectives' => [
                'Identify and hire for critical roles',
                'Establish succession pipelines',
                'Launch leadership development programs',
            ],
            'headcount_targets' => $this->getPhase1HeadcountTargets($scenario),
        ]);

        // Phase 2 (6-12 months): Scale
        $this->createPhase($scenario, 2, 'Scale', 6, 6, [
            'objectives' => [
                'Build core teams',
                'Develop internal talent',
                'Establish performance metrics',
            ],
            'headcount_targets' => $this->getPhase2HeadcountTargets($scenario),
        ]);

        // Phase 3 (12+ months): Optimize
        if ($totalTimelineMonths > 12) {
            $remainingMonths = $totalTimelineMonths - 12;
            $this->createPhase($scenario, 3, 'Optimize', 12, $remainingMonths, [
                'objectives' => [
                    'Mature processes',
                    'Optimize performance',
                    'Exit legacy roles',
                ],
                'headcount_targets' => $this->getPhase3HeadcountTargets($scenario),
            ]);
        }

        // Generate tasks for each phase
        $this->generatePhaseTasks($scenario);
    }

    /**
     * Identify blockers/dependencies between tasks
     */
    public function identifyBlockers(TransformationPhase $phase): array
    {
        $blockers = [];

        foreach ($phase->tasks as $task) {
            $potentialBlockers = [];

            //Check resource conflicts
            if ($this->hasResourceConflict($task)) {
                $potentialBlockers[] = 'Resource conflict with other tasks';
            }

            // Check skill availability
            if ($this->hasSkillBottleneck($task, $phase->scenario)) {
                $potentialBlockers[] = 'Skill bottleneck - insufficient trained people';
            }

            // Check budget constraints
            if ($this->exceeedsBudget($task, $phase->scenario)) {
                $potentialBlockers[] = 'Budget constraint - phase budget exceeded';
            }

            // Check timeline feasibility
            if ($this->isTimelineUnrealistic($task, $phase)) {
                $potentialBlockers[] = 'Timeline unrealistic given dependencies';
            }

            if (count($potentialBlockers) > 0) {
                $blockers[] = [
                    'task_id' => $task->id,
                    'task_name' => $task->task_name,
                    'blockers' => $potentialBlockers,
                ];
            }
        }

        return $blockers;
    }

    private function getPhase1HeadcountTargets(Scenario $scenario): array { ... }
    private function generatePhaseTasks(Scenario $scenario): void { ... }
    private function hasResourceConflict(TransformationTask $task): bool { ... }
    private function hasSkillBottleneck(TransformationTask $task, Scenario $scenario): bool { ... }
}
